CREATE TABLE convenios
(
  conv_id serial NOT NULL,
  conv_tipo character varying(30),
  CONSTRAINT convenios_pkey PRIMARY KEY (conv_id)
)

CREATE TABLE pacientes
(
  pac_id serial NOT NULL,
  pac_nome character varying(60),
  pac_tel character varying(20),
  pac_doenca character varying(60),
  conv_tipo character varying(30),
  CONSTRAINT pacientes_pkey PRIMARY KEY (pac_id)
)

CREATE TABLE permissoes
(
  perm_id serial NOT NULL,
  usu_id integer,
  perm_modulo character varying(60),
  perm_submodulo character varying(60),
  CONSTRAINT permissoes_pkey PRIMARY KEY (perm_id)
)

CREATE TABLE semanal
(
  sem_data date NOT NULL,
  pac_id integer NOT NULL,
  sem_pagmto boolean,
  conv_tipo character varying(30),
  CONSTRAINT semanal_pkey PRIMARY KEY (sem_data, pac_id)
)

CREATE TABLE sess_sem
(
  sess_sem_id serial NOT NULL,
  sem_data date,
  pac_id integer,
  sess_tipo character varying(60),
  CONSTRAINT sess_sem_pkey PRIMARY KEY (sess_sem_id)
)

CREATE TABLE sessoes
(
  sess_id serial NOT NULL,
  sess_tipo character varying(60),
  CONSTRAINT sessoes_pkey PRIMARY KEY (sess_id)
)

CREATE TABLE usuario
(
  usu_id integer NOT NULL DEFAULT nextval('usuario_id_usu_seq'::regclass),
  usu_nome character varying(60),
  usu_tel character varying(20),
  usu_senha character varying(12),
  CONSTRAINT usuario_pkey PRIMARY KEY (usu_id)
)
